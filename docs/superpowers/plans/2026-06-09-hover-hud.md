> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Unify hover HUD logic into a `useChartHover` hook, migrate LoudnessPanel and SpectrumPanel to use it, and add new hover HUDs to WaveformPanel and SpectrogramPanel.

**Architecture:** A new `useChartHover(computeFn)` hook holds local hover state per panel — each panel passes its own `(xFrac, yFrac) => data | null` compute function. `hoverMath.js` gains two new pure functions. `useHoverState.js` is deleted and AudioDataContext loses all hover-related fields.

**Tech Stack:** React (useState, useCallback, useRef), Vitest for unit tests, Tailwind CSS for HUD styling.

## Task 1: Create `useChartHover` hook

**Files:**
- Create: `src/hooks/useChartHover.js`

- [ ] **Step 1: Create the file**

```js
import { useCallback, useRef, useState } from "react";

export function useChartHover(computeFn) {
  const computeRef = useRef(computeFn);
  computeRef.current = computeFn;

  const [hover, setHover] = useState(null);

  const onMove = useCallback((clientX, clientY, rect) => {
    const w = Math.max(1, rect.width);
    const h = Math.max(1, rect.height);
    const xFrac = Math.max(0, Math.min(1, (clientX - rect.left) / w));
    const yFrac = Math.max(0, Math.min(1, (clientY - rect.top) / h));
    setHover(computeRef.current(xFrac, yFrac));
  }, []);

  const onLeave = useCallback(() => setHover(null), []);

  return { hover, onMove, onLeave };
}
```

```js
/**
 * @param {number} xFrac  Normalised horizontal position [0, 1]; 0 = left (oldest visible), 1 = right (newest visible)
 * @param {{ m: number, st: number }[]} histSourceList
 * @param {number} effectiveOffsetSamples
 * @param {number} visibleSamples
 * @param {number} sampleSec
 */
export function computeHistoryHoverPoint(
  xFrac,
  histSourceList,
  effectiveOffsetSamples,
  visibleSamples,
  sampleSec
) {
  if (!histSourceList.length) return null;
  const normalized = 1 - xFrac;
  const fromEndSamples = effectiveOffsetSamples + normalized * Math.max(0, visibleSamples - 1);
  const hoverIndex = Math.max(
    0,
    Math.min(histSourceList.length - 1, histSourceList.length - 1 - Math.round(fromEndSamples))
  );
  const point = histSourceList[hoverIndex];
  if (!point) return null;
  const offsetSec = Math.max(0, (histSourceList.length - 1 - hoverIndex) * sampleSec);
  const yValue = Number.isFinite(point.st) ? point.st : point.m;
  return {
    leftPct: xFrac * 100,
    topPct: Number.isFinite(yValue) ? loudnessFromTopFrac(yValue) * 100 : null,
    momentary: Number.isFinite(point.m) ? point.m : null,
    shortTerm: Number.isFinite(point.st) ? point.st : null,
    offsetLabel: formatHoverOffset(offsetSec),
  };
}
```

```js
/**
 * Finds the nearest spectrum band index to a normalised horizontal position.
 * @param {number} xFrac  [0, 1]
 * @param {{ fCenter: number }[]} bands
 * @returns {number}
 */
export function computeSpectrumHoverIndex(xFrac, bands) {
  let nearestIdx = 0;
  let nearestDist = Infinity;
  for (let i = 0; i < bands.length; i += 1) {
    const dist = Math.abs(freqToXFrac(bands[i].fCenter) - xFrac);
    if (dist < nearestDist) {
      nearestDist = dist;
      nearestIdx = i;
    }
  }
  return nearestIdx;
}
```

```js
/**
 * @param {number} xFrac  [0,1]; 0 = left (oldest visible), 1 = right (newest visible)
 * @param {number[][]} mins  mins[ch][i] — linear amplitude, per-channel, per-entry
 * @param {number[][]} maxes  maxes[ch][i]
 * @param {number} entryCount  number of visible entries
 * @param {number} effectiveOffsetSamples
 * @param {number} visibleSamples
 * @param {number} sampleSec
 * @param {string[]} labels  channel label per index
 * @returns {{ leftPct: number, timeLabel: string, channels: { label: string, dbFs: number }[] } | null}
 */
export function computeWaveformHoverPoint(
  xFrac,
  mins,
  maxes,
  entryCount,
  effectiveOffsetSamples,
  visibleSamples,
  sampleSec,
  labels
) {
  if (!entryCount || !maxes || !maxes.length) return null;
  const dataIdx = Math.max(0, Math.min(entryCount - 1, Math.round(xFrac * (entryCount - 1))));
  const stepsFromNewest = entryCount - 1 - dataIdx;
  const offsetFromNow = (stepsFromNewest + effectiveOffsetSamples) * sampleSec;
  const channels = maxes.map((chMaxes, ch) => {
    const peakAmp = Math.max(
      Math.abs(chMaxes[dataIdx] ?? 0),
      Math.abs((mins[ch]?.[dataIdx]) ?? 0)
    );
    return {
      label: labels[ch] ?? `Ch${ch + 1}`,
      dbFs: 20 * Math.log10(Math.max(1e-9, peakAmp)),
    };
  });
  return {
    leftPct: xFrac * 100,
    timeLabel: formatHoverOffset(offsetFromNow),
    channels,
  };
}
```

- [ ] **Step 4: Implement `computeSpectrogramHoverPoint` in `hoverMath.js`**

```js
/**
 * @param {number} xFrac  [0,1]; 0 = left (oldest visible), 1 = right (newest visible)
 * @param {number} yFrac  [0,1]; 0 = top (20 kHz), 1 = bottom (20 Hz)
 * @param {{ bands: { fCenter: number }[], dbList: number[] }[]} snaps
 * @param {number} effectiveOffsetSamples  in visual snap units
 * @param {number} visibleSamples          in visual snap units
 * @param {number} sampleSec               seconds per visual snap
 * @returns {{ leftPct, topPct, timeLabel, freqLabel, dbLabel } | null}
 */
export function computeSpectrogramHoverPoint(
  xFrac,
  yFrac,
  snaps,
  effectiveOffsetSamples,
  visibleSamples,
  sampleSec
) {
  if (!snaps || !snaps.length) return null;
  const total = snaps.length;
  const newestVisible = total - 1 - effectiveOffsetSamples;
  const oldestVisible = newestVisible - visibleSamples + 1;
  const visStart = Math.max(0, Math.floor(oldestVisible));
  const visEnd = Math.max(0, Math.min(total - 1, Math.floor(newestVisible)));
  if (visEnd < visStart) return null;
  const snapIdx = Math.max(
    visStart,
    Math.min(visEnd, Math.round(visStart + xFrac * (visEnd - visStart)))
  );
  const snap = snaps[snapIdx];
  if (!snap?.dbList?.length || !snap?.bands?.length) return null;
  const offsetFromNow = Math.max(0, (total - 1 - snapIdx) * sampleSec);
  const hz = hzFromFrac(yFrac);
  const bands = snap.bands;
  let lo = 0;
  let hi = bands.length - 1;
  while (lo < hi) {
    const mid = (lo + hi) >> 1;
    if (bands[mid].fCenter < hz) lo = mid + 1;
    else hi = mid;
  }
  if (
    lo > 0 &&
    Math.abs(Math.log(bands[lo - 1].fCenter) - Math.log(hz)) <
      Math.abs(Math.log(bands[lo].fCenter) - Math.log(hz))
  ) {
    lo--;
  }
  const db = snap.dbList[lo];
  return {
    leftPct: xFrac * 100,
    topPct: yFrac * 100,
    timeLabel: formatHoverOffset(offsetFromNow),
    freqLabel: formatSpectrumFreq(hz),
    dbLabel: Number.isFinite(db) ? `${db.toFixed(1)} dB` : "-",
  };
}
```
